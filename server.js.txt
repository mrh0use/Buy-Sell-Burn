const express = require('express');
const cors = require('cors');
const axios = require('axios');

const app = express();
const PORT = 4000;

app.use(cors());

app.get('/fetchNFTs', async (req, res) => {
    const contractAddress = req.query.contract;
    try {
        const response = await axios.get(`https://api.opensea.io/api/v1/assets?order_by=token_id&order_direction=desc&asset_contract_address=${contractAddress}&limit=3`, {
            headers: {
                'X-API-KEY': process.env.REACT_APP_OPENSEA_API_KEY
            }
        });
        res.json(response.data);
    } catch (error) {
        res.status(500).json({ error: 'Failed to fetch data from OpenSea' });
    }
});

app.listen(PORT, () => {
    console.log(`Server running on http://localhost:${PORT}`);
});
